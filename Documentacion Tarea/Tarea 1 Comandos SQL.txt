CREATE TABLE Usuario(
idUsuario INT PRIMARY KEY,
nombre VARCHAR(45),
apellidos VARCHAR(45),
edad INT,
nacionalidad VARCHAR(45),
url_foto VARCHAR(200),
login VARCHAR(45),
password VARCHAR(45))



CREATE TABLE Artista(
idArtista INT PRIMARY KEY,
nombre VARCHAR(45),
nacionalidad VARCHAR(45),
genero VARCHAR(45),
url_foto VARCHAR(200))



CREATE TABLE Lista(
idLista INT PRIMARY KEY,
nombre VARCHAR(45),
descripcion VARCHAR(200))



CREATE TABLE Disco(
idDisco INT PRIMARY KEY,
Artista_idArtista INT REFERENCES Artista,
nombre VARCHAR(45),
año VARCHAR(45),
url_foto VARCHAR(45))



CREATE TABLE Sigue(
idSigue INT PRIMARY KEY,
Usuario_idUsuario INT REFERENCES Usuario,
Artista_idArtista INT REFERENCES Artista,
Disco_idDisco INT REFERENCES Disco,
Lista_idLista INT REFERENCES Lista)



CREATE TABLE Canción(
idCanción INT PRIMARY KEY,
Disco_idDisco INT REFERENCES Disco,
nombre VARCHAR(45),
duracion VARCHAR(45),
numero_pista VARCHAR(45),
link VARCHAR(200),
popularidad VARCHAR(200))



CREATE TABLE listaCanción(
idlistaCanción INT PRIMARY KEY,
Canción_idCanción INT REFERENCES Canción,
Lista_idLista INT REFERENCES Lista)
