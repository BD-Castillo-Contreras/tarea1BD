create sequence idusuario_secuencia
start with 1 
increment by 1 
nomaxvalue; 

create trigger incrementoidusuario_trigger
before insert on usuario
for each row
begin
select idusuario_secuencia.nextval into :new.idusuario from dual;
end;
/
